#include <stdio.h>

int main() {
    int count = 0, num = 2, sum = 0, i, isPrime;
    while (count < 10) {
        isPrime = 1;  
        for (i = 2; i <= num / 2; i++) {
            if (num % i == 0) {
                isPrime = 0;
                break;
            }
        }
        if (isPrime) {
            sum += num;
            count++;
        }
        num++;
    }
    printf("Sum of first 10 prime numbers = %d\n", sum);
    return 0;
}
