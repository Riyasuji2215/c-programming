#include <stdio.h>
#include <string.h>
int compareStrings(char str1[], char str2[]) {
    int i = 0;
    while (str1[i] != '\0' && str2[i] != '\0') {
        if (str1[i] != str2[i]) {
            return str1[i] - str2[i];  
        }
        i++;
    }
    return str1[i] - str2[i];
}
int main() {
    char str1[100], str2[100];
    int result1, result2;
    printf("Enter first string: ");
    scanf("%s", str1);
    printf("Enter second string: ");
    scanf("%s", str2);
    result1 = strcmp(str1, str2);
    if (result1 == 0)
        printf("\n[strcmp] Both strings are equal.\n");
    else if (result1 < 0)
        printf("\n[strcmp] First string is smaller.\n");
    else
        printf("\n[strcmp] First string is greater.\n");
    result2 = compareStrings(str1, str2);
    if (result2 == 0)
        printf("[manual] Both strings are equal.\n");
    else if (result2 < 0)
        printf("[manual] First string is smaller.\n");
    else
        printf("[manual] First string is greater.\n");

    return 0;
}
