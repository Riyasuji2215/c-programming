code 3 do while 

#include <stdio.h>
int main() {
    int i = 1;
    do {
        printf("Number: %d\n", i);
        i++;
    } while (i <= 5);
    return 0;
}

code 4 switch statment

#include <stdio.h>
int main() {
    int choice;
    printf("Enter a number between 1 and 4: ");
    scanf("%d", &choice);
    switch(choice) {
        case 1:
            printf("You chose One\n");
            break;
        case 2:
            printf("You chose Two\n");
            break;
        case 3:
            printf("You chose Three\n");
            break;
        case 4:
            printf("You chose Four\n");
            break;
        default:
            printf("Invalid choice! Please enter between 1 and 4.\n");
    }
    return 0;
}


O/P

Enter a number between 1 and 4: 2
You chose Two
