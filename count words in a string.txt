#include <stdio.h>
#include <string.h>
#include <ctype.h>
int main() {
    char str[200];
    int i, words = 0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    int len = strlen(str);
    for (i = 0; i < len; i++) {
        if ((i == 0 && !isspace(str[i])) || 
            (isspace(str[i]) && !isspace(str[i + 1]) && str[i + 1] != '\0')) {
            words++;
        }
    }
    printf("Total words: %d\n", words);
    return 0;
}
